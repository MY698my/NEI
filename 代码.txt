import os
import re
import pandas as pd
from dotenv import load_dotenv
from PyPDF2 import PdfReader
from langchain_openai import ChatOpenAI
from langchain.agents import AgentExecutor, create_openai_tools_agent
from langchain.tools import Tool
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder

# ===================== 1. 基础配置 =====================
# 加载环境变量（请在同级目录创建 .env 文件，填写 OPENAI_API_KEY）
load_dotenv()
os.environ["OPENAI_API_KEY"] = os.getenv("OPENAI_API_KEY")

# 初始化大模型（可替换为通义千问/文心一言/本地Llama）
llm = ChatOpenAI(model="gpt-3.5-turbo", temperature=0.1)

# ===================== 2. 公共工具函数 =====================
def read_pdf(pdf_path: str) -> str:
    """读取PDF文献，返回纯文本"""
    try:
        reader = PdfReader(pdf_path)
        text = ""
        for page in reader.pages:
            text += page.extract_text()
        return text[:10000]  # 截断文本避免超限
    except Exception as e:
        return f"PDF读取失败：{str(e)}"

def extract_entities(text: str) -> dict:
    """从文本提取疾病、基因、通路等核心实体（生信专用）"""
    pattern_disease = re.compile(r'(cancer|tumor|diabetes|hypertension|OHP|ovarian cancer)', re.I)
    pattern_gene = re.compile(r'[A-Z]{2,}[0-9]?')  # 基因命名规则
    diseases = list(set(pattern_disease.findall(text)))
    genes = list(set(pattern_gene.findall(text)))[:10]
    return {"疾病": diseases, "基因": genes, "通路": ["PI3K-Akt", "MAPK", "Wnt"]}  # 预设生信通路

# 封装为LangChain工具
tools = [
    Tool(name="读取PDF文献", func=read_pdf, description="输入PDF文件路径，返回文献文本内容"),
    Tool(name="提取生信实体", func=extract_entities, description="输入文本，提取疾病、基因、信号通路实体")
]

# ===================== 3. 四大核心Agent定义 =====================
# 通用Agent模板
def create_research_agent(role_prompt: str):
    prompt = ChatPromptTemplate.from_messages([
        ("system", role_prompt),
        ("user", "{input}"),
        MessagesPlaceholder(variable_name="agent_scratchpad"),
    ])
    agent = create_openai_tools_agent(llm, tools, prompt)
    return AgentExecutor(agent=agent, tools=tools, verbose=False)

# 1. 文献解析Agent：批量解析PDF，提取核心数据
agent1_parse = create_research_agent(
    "你是专业的生信文献解析专家，负责读取PDF文献，提取研究目的、实验数据、核心结论"
)

# 2. 知识图谱Agent：构建疾病-基因-通路关联关系
agent2_knowledge = create_research_agent(
    "你是知识图谱构建专家，基于提取的实体，梳理疾病、基因、通路的关联关系"
)

# 3. 推理验证Agent：生成科研假设，生成分析脚本
agent3_reason = create_research_agent(
    "你是生信科研推理专家，基于关联图谱生成可验证的科学假设，输出GEO数据集分析脚本"
)

# 4. 报告生成Agent：自动生成科研调研报告
agent4_report = create_research_agent(
    "你是科研报告撰写专家，整合所有结果，生成结构化的课题调研初稿"
)

# ===================== 4. 多Agent协作主流程 =====================
def research_assistant_main(pdf_path: str):
    print("="*50)
    print("🚀 多Agent科研文献挖掘助手启动")
    print("="*50)

    # Step1：文献解析Agent执行
    print("\n📄 【Agent1】文献解析中...")
    parse_result = agent1_parse.invoke({"input": f"解析这个文献：{pdf_path}，提取核心内容"})
    print("✅ 文献解析完成")

    # Step2：知识图谱Agent执行
    print("\n🔗 【Agent2】构建知识图谱中...")
    entity_data = extract_entities(parse_result["output"])
    knowledge_result = agent2_knowledge.invoke({"input": f"构建关联关系：{entity_data}"})
    print("✅ 知识图谱构建完成")

    # Step3：推理验证Agent执行
    print("\n🧠 【Agent3】生成科研假设中...")
    reason_result = agent3_reason.invoke({"input": f"基于{knowledge_result}生成科研假设+GEO分析脚本"})
    print("✅ 科研假设生成完成")

    # Step4：报告生成Agent执行
    print("\n📝 【Agent4】生成科研报告中...")
    final_report = agent4_report.invoke({
        "input": f"整合以下内容生成报告：\n文献解析：{parse_result}\n知识图谱：{knowledge_result}\n假设：{reason_result}"
    })

    # 保存最终报告
    with open("科研文献分析报告.md", "w", encoding="utf-8") as f:
        f.write(final_report["output"])

    print("\n🎉 全部执行完成！报告已保存为：科研文献分析报告.md")
    print("="*50)
    return final_report["output"]

# ===================== 5. 运行入口 =====================
if __name__ == "__main__":
    # 替换为你的科研PDF路径（支持生信/医学PDF）
    YOUR_PDF_PATH = "research_paper.pdf"
    research_assistant_main(YOUR_PDF_PATH)
